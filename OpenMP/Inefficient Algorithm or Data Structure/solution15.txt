#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define NUM_ELEMENTS 100000

int <FUNC_NAME>() {
    int* <PARAM_1> = (int*)malloc(NUM_ELEMENTS * sizeof(int));

    // Populate the array
    for (int i = 0; i < NUM_ELEMENTS; i++) {
        <PARAM_1>[i] = i;
    }

    // Access elements randomly
    #pragma omp parallel for
    for (int i = 0; i < NUM_ELEMENTS; i++) {
        int index = rand() % NUM_ELEMENTS;
        int data = <PARAM_1>[index];
        printf("Thread %d: Accessed element at index %d with value %d\n", omp_get_thread_num(), index, data);
    }

    free(<PARAM_1>);

    return 0;
}
