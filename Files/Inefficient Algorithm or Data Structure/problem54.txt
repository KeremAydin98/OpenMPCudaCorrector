#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define ARRAY_SIZE 100000

void insertElement(int* array, int index, int element) {
    for (int i = ARRAY_SIZE - 1; i > index; i--) {
        array[i] = array[i - 1];
    }
    array[index] = element;
}

void deleteElement(int* array, int index) {
    for (int i = index; i < ARRAY_SIZE - 1; i++) {
        array[i] = array[i + 1];
    }
    array[ARRAY_SIZE - 1] = 0;  // Reset the last element
}

int main() {
    int* array = (int*)malloc(ARRAY_SIZE * sizeof(int));

    // Initialize the array
    for (int i = 0; i < ARRAY_SIZE; i++) {
        array[i] = i;
    }

    int insertIndex = ARRAY_SIZE / 2;  // Insert at the middle
    int deleteIndex = ARRAY_SIZE / 4;  // Delete from the quarter

    // Insert an element
    insertElement(array, insertIndex, -1);

    // Delete an element
    deleteElement(array, deleteIndex);

    // Print the array
    for (int i = 0; i < ARRAY_SIZE; i++) {
        printf("%d ", array[i]);
    }
    printf("\n");

    free(array);

    return 0;
}
