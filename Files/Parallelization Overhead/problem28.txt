#include <stdio.h>
#include <omp.h>

#define ARRAY_SIZE 1000000

int main() {
    int i;
    int array[ARRAY_SIZE];
    int sum = 0;

    // Initialize the array
    for (i = 0; i < ARRAY_SIZE; i++) {
        array[i] = i + 1;
    }

    #pragma omp parallel for reduction(+:sum)
    for (i = 0; i < ARRAY_SIZE; i++) {
        sum += array[i];
    }

    printf("Sum: %d\n", sum);

    return 0;
}
