#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define SIZE 1000000

int main() {
    int* array = (int*) malloc(SIZE * sizeof(int));
    int* temp = (int*) malloc(SIZE * sizeof(int));

    // Initialize the array
    for (int i = 0; i < SIZE; i++) {
        array[i] = i;
    }

    #pragma omp parallel
    {
        #pragma omp for
        for (int i = 0; i < SIZE; i++) {
            // Perform some computation on the array
            temp[i] = array[i] * 2;
        }

        #pragma omp for
        for (int i = 0; i < SIZE; i++) {
            // Update the original array with the computed values
            array[i] = temp[i];
        }
    }

    free(temp);
    free(array);

    return 0;
}
