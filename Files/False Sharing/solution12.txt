#include <cstdio>
#include <iostream>
#include <omp.h>

int <VAR_4> = 10000000;
const int <VAR_PAD> = 16;

int main()
{

  double start_time = omp_get_wtime();

  double <PARAM_1>[<VAR_4> + <VAR_PAD>],  <PARAM_2>[<VAR_4> + <VAR_PAD>], <VAR_2>[<VAR_4> + <VAR_PAD>];

#pragma omp parallel
{

#pragma omp for
  for(int i=0;i<<VAR_4>;i++){
    <PARAM_1>[i] = 10;
  }

#pragma omp for
  for(int i=0;i<<VAR_4>;i++){
    <PARAM_2>[i] = 10;
  }

#pragma omp for
  for(int i=0;i<<VAR_4>;i++){
    <VAR_2>[i] = <PARAM_1>[i] + <PARAM_2>[i];
  }

}	
  double end_time = omp_get_wtime();

  printf("Elapsed time: %f", end_time - start_time);

  return 0;
}
