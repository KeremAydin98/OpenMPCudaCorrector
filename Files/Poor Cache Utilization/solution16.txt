#include <iostream>
#include <omp.h>

using namespace std;

int <FUNC_NAME>() {
    const int N = 1000000;
    int arr[N];

    #pragma omp parallel for schedule(static)
    for(int i = 0; i < N; i++) {
        arr[i] = i;
    }

    int sum = 0;
    #pragma omp parallel for schedule(static) reduction(+:sum)
    for(int i = 0; i < N; i++) {
        sum += arr[i];
    }

    cout << "Sum: " << sum << endl;

    return 0;
}
