int countArrScalarPar_01(int arr[], int scalar){
	int nthreads = 16;
	double total = 0;
	double s_priv[nthreads] = {0};

	#pragma omp parallel num_threads(nthreads)
	{
		int t = omp_get_thread_num();
		#pragma omp for
		for (int i = 0; i < ARRAY_SIZE; i++){
			if (arr[i] == scalar){
				s_priv[t]++;
			}
		}	
	}

	for (int i = 0; i < nthreads; i++)
	{
		total += s_priv[i];
	}

	return total;
}
