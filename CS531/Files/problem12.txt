#include <cstdio>
#include <iostream>
#include <omp.h>

int MAX = 10000000;

int main()
{

  double start_time = omp_get_wtime();

  double res[MAX],  kes[MAX], result[MAX];

#pragma omp parallel
{

#pragma omp for
  for(int i=0;i<MAX;i++){
    res[i] = 10;
  }

#pragma omp for
  for(int i=0;i<MAX;i++){
    kes[i] = 10;
  }

#pragma omp for
  for(int i=0;i<MAX;i++){
    result[i] = res[i] + kes[i];
  }

}	
  double end_time = omp_get_wtime();

  printf("Elapsed time: %f", end_time - start_time);

  return 0;
}
