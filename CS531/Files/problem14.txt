#include <iostream>
#include <omp.h>

const int ARRAY_SIZE = 10000;
const int NUM_THREADS = 4;

int main() {
    int data1[ARRAY_SIZE];
    int data2[ARRAY_SIZE];

    // Initialize data arrays
    for (int i = 0; i < ARRAY_SIZE; i++) {
        data1[i] = i;
        data2[i] = i;
    }

    // Parallelize array update using OpenMP
    #pragma omp parallel num_threads(NUM_THREADS)
    {
        int thread_id = omp_get_thread_num();

        // Divide array into equal parts for each thread
        int chunk_size = ARRAY_SIZE / NUM_THREADS;
        int start_index = thread_id * chunk_size;
        int end_index = start_index + chunk_size;

        // Update each thread's portion of the arrays
        for (int i = start_index; i < end_index; i++) {
            // Access variables located on different cache lines
            data1[i] += thread_id;
            data2[i] += thread_id;
        }
    }

    // Print out updated arrays
    for (int i = 0; i < ARRAY_SIZE; i++) {
        std::cout << data1[i] << " ";
    }
    std::cout << std::endl;

    for (int i = 0; i < ARRAY_SIZE; i++) {
        std::cout << data2[i] << " ";
    }
    std::cout << std::endl;

    return 0;
}
